def jarakData(i, j):
  P = 3
  M = 0
  for data in ['race/ethnicity', 'parental level of education', 'lunch' ]:
    if dataset[data][i] == dataset[data][j]:
      M = M + 1
  return (P-M)/P

print(jarakData(0,1))
print(jarakData(0,2))
print(jarakData(0,3))
